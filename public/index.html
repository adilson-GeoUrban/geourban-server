// 🔐 ALTERAR SENHA (PROTEGIDO) app.post('/change-password', (req, res) => { const token = req.headers['authorization']; const { currentPass, newPass } = req.body; // 🚫 valida token if (!token) { return res.status(401).json({ message: "Não autorizado" }); } try { // 🔐 decode token const decoded = Buffer.from(token, 'base64').toString(); const [user, timestamp] = decoded.split("|"); if (!user || !timestamp) { return res.status(401).json({ message: "Token inválido" }); } // ⏳ valida expiração (ex: 24h) const now = Date.now(); const tokenTime = parseInt(timestamp); if (isNaN(tokenTime) || (now - tokenTime > 86400000)) { return res.status(401).json({ message: "Token expirado" }); } // 🧠 banco simulado if (!global.users) global.users = {}; const storedPass = global.users[user]; if (!storedPass) { return res.status(404).json({ message: "Usuário não encontrado" }); } // 🔒 valida senha atual if (storedPass !== currentPass) { return res.status(400).json({ message: "Senha atual incorreta" }); } // 🚫 evitar mesma senha if (currentPass === newPass) { return res.status(400).json({ message: "Nova senha não pode ser igual à atual" }); } // 🔐 força mínima if (!newPass || newPass.length < 6) { return res.status(400).json({ message: "Nova senha muito fraca" }); } // ✅ atualiza senha global.users[user] = newPass; console.log(`[SECURITY] ${user} alterou senha em ${new Date().toISOString()}`); res.json({ message: "Senha alterada com sucesso" }); } catch (err) { console.error("[ERRO CHANGE PASSWORD]", err); res.status(500).json({ message: "Erro interno" }); } });
