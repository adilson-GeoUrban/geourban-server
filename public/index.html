// 🚀 SISTEMA GEOURBAN — BLOCO ÚNICO const express = require("express"); const app = express(); app.use(express.json()); // 🛡️ IA PROTEÇÃO (GUARDIAN) function guardian(texto) { const proibidos = [ "hack", "fraude", "ataque", "pornografia", "racismo", "ódio" ]; texto = (texto || "").toLowerCase(); for (let p of proibidos) { if (texto.includes(p)) { return { bloqueado: true }; } } return { bloqueado: false }; } // ⚖️ GOVERNANÇA function governar(texto) { if (texto.includes("erro")) { return { permitido: false }; } return { permitido: true }; } // 🧠 IA PRINCIPAL function sistema(dados) { // 🔴 PROTEÇÃO const g = guardian(dados.descricao); if (g.bloqueado) { return { status: "bloqueado", motivo: "Conteúdo proibido" }; } // 🟡 GOVERNANÇA const gov = governar(dados.descricao); if (!gov.permitido) { return { status: "bloqueado", motivo: "Violação de regra" }; } // 🟢 EXECUÇÃO return { status: "ok", resposta: "Sistema funcionando 🚀", dados: dados }; } // 💾 BACKUP const fs = require("fs"); function salvarBackup(dados) { const nome = `backup-${Date.now()}.json`; fs.writeFileSync(nome, JSON.stringify(dados, null, 2)); } // 📡 ROTA PRINCIPAL app.post("/solicitar", (req, res) => { const resultado = sistema(req.body); // salvar backup sempre salvarBackup(req.body); if (resultado.status === "bloqueado") { return res.status(403).json(resultado); } res.json(resultado); }); // 🌐 ROTA TESTE app.get("/", (req, res) => { res.send("GeoUrban rodando 🚀"); }); // ▶️ INICIAR app.listen(3000, () => { console.log("Servidor rodando em http://localhost:3000"); });
