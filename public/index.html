function definirCamadas() { const perfil = localStorage.getItem("perfil_usuario"); const objetivo = (localStorage.getItem("objetivo") || "").toLowerCase(); let camadasBase = []; let extras = []; // 🎓 ESTUDANTES if (perfil === "estudantes") { camadasBase = ["educacao", "cursos", "simulacoes"]; } // 👷 PROFISSIONAIS if (perfil === "profissional") { camadasBase = ["consultoria", "clientes", "cadastro_tecnico", "loja"]; } // 🏛️ GOVERNO if (perfil === "governo") { camadasBase = ["gestao", "auditoria", "relatorios", "controle"]; } // 🎯 OBJETIVO refina sem duplicar if (objetivo.includes("consultoria")) extras.push("consultoria"); if (objetivo.includes("curso")) extras.push("cursos"); if (objetivo.includes("loja")) extras.push("loja"); // 🔥 remove duplicados automaticamente const camadas = [...new Set([...camadasBase, ...extras])]; localStorage.setItem("camadas", JSON.stringify(camadas)); escrever("Luiza: Estrutura personalizada ativada. 🧠"); escrever("Luiza: Camadas liberadas: " + camadas.join(", ")); } profissional engenheiro consultoria consultoria, clientes, cadastro_tecnico, loja
