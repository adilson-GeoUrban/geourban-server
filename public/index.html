function adicionarMensagemNaTela(msg) { const chat = document.getElementById("chat"); const p = document.createElement("p"); p.innerText = msg; // 🔒 seguro contra XSS chat.appendChild(p); chat.scrollTop = chat.scrollHeight; } async function enviar() { const inputEl = document.getElementById("input"); const texto = inputEl.value.trim(); if (!texto) return; adicionarMensagemNaTela("Você: " + texto); inputEl.value = ""; // 🤖 efeito digitando adicionarMensagemNaTela("Luiza está digitando..."); try { const controller = new AbortController(); const timeout = setTimeout(() => controller.abort(), 5000); const res = await fetch("/ia", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ mensagem: texto }), signal: controller.signal }); clearTimeout(timeout); const data = await res.json(); const chat = document.getElementById("chat"); // 🔒 CORREÇÃO PRINCIPAL (compatível com backend antigo ou novo) const mensagem = data?.mensagem || data?.resposta; if (!mensagem) { chat.lastChild.innerText = "Sistema: resposta inválida"; return; } // substitui "digitando..." chat.lastChild.innerText = "Luiza: " + mensagem; // 🚀 REDIRECT GARANTIDO if (data.acao === "REDIRECT" && data.destino) { setTimeout(() => { window.location.href = data.destino; }, 800); } } catch (erro) { const chat = document.getElementById("chat"); chat.lastChild.innerText = "Erro ao conectar com servidor"; } } // ⌨️ ENTER envia mensagem document.getElementById("input").addEventListener("keypress", function(e) { if (e.key === "Enter") { enviar(); } });
