async function login() { const email = document.getElementById("email").value; const password = document.getElementById("password").value; const status = document.getElementById("status"); status.innerText = "Conectando..."; try { const res = await fetch("/login", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ email, password }) }); const data = await res.json(); if (!data.success) { status.innerText = "Erro login"; return; } localStorage.setItem("token", data.token); localStorage.setItem("usuario", data.user.email); window.location.href = "/ia"; } catch (e) { status.innerText = "Erro servidor"; } }
