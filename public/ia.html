const express = require("express"); const path = require("path"); const app = express(); app.use(express.json()); // 🔥 ESSENCIAL app.use(express.static(path.join(__dirname, "public"))); app.get("/", (req, res) => { res.sendFile(path.join(__dirname, "public", "index.html")); }); // health check app.get("/health", (req, res) => { res.json({ status: "ok" }); }); const PORT = process.env.PORT; app.listen(PORT, "0.0.0.0", () => { console.log("Rodando na porta", PORT); });
