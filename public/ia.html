async function goPanel() { try { const res = await fetch("/ia.html", { method: "HEAD", cache: "no-store" }); // ⚠️ alguns hosts retornam 403/404 mesmo com arquivo válido if (res.ok || res.status === 0 || res.status === 200) { window.location.replace("/ia.html"); } else { window.location.replace("/"); } } catch (e) { console.error("Erro ao acessar painel:", e); window.location.replace("/"); } }
