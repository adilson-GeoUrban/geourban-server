async function goPanel() { try { const res = await fetch("/ia.html", { method: "HEAD", cache: "no-store" }); // ✔ considera sucesso real do servidor const ok = res && (res.ok || (res.status >= 200 && res.status < 300)); if (ok) { window.location.replace("/ia.html"); } else { window.location.replace("/"); } } catch (e) { console.error("Erro ao acessar painel:", e); window.location.replace("/"); } }
