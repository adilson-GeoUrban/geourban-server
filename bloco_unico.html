// 🚀 BD2 - BACKUP AUTOMÁTICO MELHORADO function backup(dados) { try { const json = JSON.stringify(dados, null, 2); const blob = new Blob([json], { type: "application/json" }); const url = URL.createObjectURL(blob); const a = document.createElement("a"); a.href = url; // nome com data (controle real) const data = new Date().toISOString().slice(0,10); a.download = `backup_geourban_${data}.json`; document.body.appendChild(a); a.click(); // limpeza (EVITA TRAVAMENTO) document.body.removeChild(a); URL.revokeObjectURL(url); console.log("✅ Backup OK"); } catch (erro) { console.error("❌ Erro backup:", erro); alert("Erro ao gerar backup"); } }
