const express = require("express"); const jwt = require("jsonwebtoken"); const bcrypt = require("bcryptjs"); const app = express(); app.use(express.json()); // 🔐 CONFIG const SECRET = "geo_urban_super_chave_segura"; // 👤 USUÁRIO (senha criptografada) const senhaHash = bcrypt.hashSync("1234", 8); let usuarios = [ { user: "admin", pass: senhaHash } ]; // 💾 DADOS let financeiro = []; let logs = []; // 🧠 LOG DE SEGURANÇA function registrarLog(tipo, info) { logs.push({ tipo, info, data: new Date().toISOString() }); } // 🔐 MIDDLEWARE DE PROTEÇÃO function verificarToken(req, res, next) { const token = req.headers["authorization"]; if (!token) { registrarLog("ERRO", "Token ausente"); return res.status(403).json({ erro: "Sem token" }); } jwt.verify(token, SECRET, (err, decoded) => { if (err) { registrarLog("ERRO", "Token inválido"); return res.status(401).json({ erro: "Token inválido" }); } req.user = decoded; next(); }); } // 🌐 FRONTEND EMBUTIDO const html = ` GeoUrban Secure

Login Seguro



Painel Seguro

`; // 🌐 ROTA PRINCIPAL app.get("/", (req, res) => res.send(html)); // 🔐 LOGIN app.post("/login", async (req, res) => { const { user, pass } = req.body; const usuario = usuarios.find(u => u.user === user); if (!usuario) { registrarLog("ERRO", "Usuário não encontrado"); return res.status(401).json({ erro: "Login inválido" }); } const valido = await bcrypt.compare(pass, usuario.pass); if (!valido) { registrarLog("ERRO", "Senha incorreta"); return res.status(401).json({ erro: "Senha inválida" }); } const token = jwt.sign({ user }, SECRET, { expiresIn: "1h" }); registrarLog("SUCESSO", "Login realizado"); res.json({ token }); }); // 💾 FINANCEIRO (PROTEGIDO) app.post("/financeiro", verificarToken, (req, res) => { financeiro.push(req.body); registrarLog("INFO", "Registro financeiro adicionado"); res.json({ ok: true }); }); app.get("/financeiro", verificarToken, (req, res) => { res.json(financeiro); }); // 📊 LOGS (PROTEGIDO) app.get("/logs", verificarToken, (req, res) => { res.json(logs); }); // 🚀 START app.listen(3000, () => { console.log("🔐 Sistema seguro rodando em http://localhost:3000"); });
